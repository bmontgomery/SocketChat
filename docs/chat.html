Sends a message to all users connected to the Chat Server except for the user from which the message was sent.

 
	ChatServer.prototype.publishMessage = function(sender, message) {
		this.users.forEach(function(user) {
			if (user != sender) {
				user.sendMessage(message);
			}
		});
	}
